# Make “Normal” paragraphs plain <p>
p[style-name='Normal'] => p

# Turn “Quote” / “Citation” into blockquote (if used)
p[style-name='Citation'] => blockquote:fresh
p[style-name='Quote'] => blockquote:fresh

# Bibliography-like lines get a hanging indent
p[style-name='Bibliographie'] => p.bib-entry

# Keep footnotes; Mammoth auto-maps them. Mark footnote references:
note-reference => sup.note-ref

# Avoid spans for bold/italic headings converted from Word
r[style-name='Strong'] => strong
r[style-name='Emphasis'] => em
