# Minimal Mammoth style map.
# Headings are mapped automatically by Mammoth:
#   "Heading 1" → <h1>, "Heading 2" → <h2>, etc.

# Keep "Normal" paragraphs as plain <p>
p[style-name='Normal'] => p

# If you use a “Citation”/“Quote” style in Word, render as <blockquote>
p[style-name='Citation'] => blockquote:fresh
p[style-name='Quote'] => blockquote:fresh

# Bibliography-like paragraphs (if such a custom style exists) with hanging indent
p[style-name='Bibliographie'] => p.bib-entry

# Footnote references as compact superscripts with a link
note-reference => sup.note-ref

# Strong/emphasis runs map cleanly
r[style-name='Strong'] => strong
r[style-name='Emphasis'] => em
